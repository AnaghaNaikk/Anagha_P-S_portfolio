Single-file script

This script reads CSVs if present (marks.csv, diamonds.csv, used_cars.csv). If files are missing it uses small example arrays. It computes mean, median, mode, variance, standard deviation, range, IQR, skewness, kurtosis, and detects outliers using the IQR rule. Requires pandas, numpy, and scipy.

# lab1_stats_examples.py
# Save this code as lab1_stats_examples.py and run:

import os
import numpy as np
import pandas as pd
from scipy import stats

def load_or_example(filename, column, example_array):
    """
    Try to load a single-column CSV. If not found, return the example array as a pandas Series.
    """
    if os.path.exists(filename):
        try:
            df = pd.read_csv(filename)
            if column in df.columns:
                return df[column].dropna().astype(float)
            else:
                print(f\"Warning: column '{column}' not found in {filename}. Using example data.\")
        except Exception as e:
            print(f\"Warning: could not read {filename}: {e}. Using example data.\")
    # fallback
    return pd.Series(example_array, name=column)

# Example datasets (small) for manual practice and verification
marks_example = [56, 72, 68, 90, 45, 77, 61, 83, 59, 95]
used_cars_example = [0.8, 1.2, 0.5, 2.5, 0.9, 1.0, 0.6, 3.5, 0.4, 1.8]  # in lakhs
diamonds_example = pd.DataFrame({
    'price': [500, 1200, 800, 2500, 400, 1500],
    'carat': [0.3, 0.9, 0.5, 1.5, 0.25, 1.0],
    'clarity': ['SI1', 'VS2', 'SI2', 'VVS1', 'I1', 'VS1']
})

# Load or use examples
marks = load_or_example('marks.csv', 'Marks', marks_example)
used_cars = load_or_example('used_cars.csv', 'Selling_Price', used_cars_example)
# For diamonds we try to load the CSV; if not present we use the small example DataFrame
if os.path.exists('diamonds.csv'):
    try:
        diamonds = pd.read_csv('diamonds.csv')
    except Exception as e:
        print('Warning: could not read diamonds.csv:', e)
        diamonds = diamonds_example.copy()
else:
    diamonds = diamonds_example.copy()

def summary_stats(series):
    """
    Compute a set of summary statistics for a numeric pandas Series.
    Returns a dict with values.
    """
    arr = series.dropna().astype(float)
    n = len(arr)
    mean = arr.mean()
    median = arr.median()
    # mode may return multiple values; pick all
    mode_vals = arr.mode().tolist()
    var_sample = arr.var(ddof=1)   # sample variance
    std_sample = arr.std(ddof=1)   # sample standard deviation
    data_range = arr.max() - arr.min()
    q1 = arr.quantile(0.25)
    q3 = arr.quantile(0.75)
    iqr = q3 - q1
    # skewness and kurtosis (sample)
    skewness = stats.skew(arr, bias=False) if n > 2 else float('nan')
    kurtosis = stats.kurtosis(arr, bias=False) if n > 3 else float('nan')
    # outliers by 1.5*IQR rule
    lower_bound = q1 - 1.5 * iqr
    upper_bound = q3 + 1.5 * iqr
    outliers = arr[(arr < lower_bound) | (arr > upper_bound)].tolist()
    return {
        'count': n,
        'mean': mean,
        'median': median,
        'mode': mode_vals,
        'variance_sample': var_sample,
        'std_sample': std_sample,
        'min': arr.min(),
        'max': arr.max(),
        'range': data_range,
        'q1': q1,
        'q3': q3,
        'iqr': iqr,
        'lower_bound': lower_bound,
        'upper_bound': upper_bound,
        'outliers': outliers,
        'skewness': skewness,
        'kurtosis': kurtosis
    }

# Example: manual calculation demonstration for mean and median (prints steps)
def manual_mean_steps(arr):
    arr = list(arr)
    s = sum(arr)
    n = len(arr)
    mean = s / n
    print('\\nManual mean calculation steps:')
    print('Values:', arr)
    print('Sum =', s)
    print('Count =', n)
    print('Mean = Sum / Count =', mean)
    return mean

def manual_median_steps(arr):
    arr = sorted(arr)
    n = len(arr)
    print('\\nManual median calculation steps:')
    print('Sorted values:', arr)
    if n % 2 == 1:
        median = arr[n//2]
        print('Odd count -> median is middle value:', median)
    else:
        median = (arr[n//2 - 1] + arr[n//2]) / 2
        print('Even count -> median is average of two middle values:', median)
    return median

# Demonstrate with the marks example (or loaded marks)

manual_median_steps(marks.tolist())

Notes & Tips

How to use these questions

Start by solving the first 4–6 questions by hand for the small example arrays. Then run the Python script to verify your answers. For larger datasets, rely on the script to compute accurate statistics and to detect outliers automatically.

When to use each measure

  • Mean — use when distribution is symmetric and you want the arithmetic average.
  • Median — use when distribution is skewed or contains outliers; it is more robust.
  • Mode — useful for categorical or discrete data to find the most frequent value.
  • Standard deviation / variance — measure spread; use sample formulas for sample data (ddof=1).
  • IQR — robust measure of spread; used for outlier detection with the 1.5×IQR rule.
